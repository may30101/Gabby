  <script>

    function fmt(n) {

      return '$' + Math.round(n).toLocaleString('zh-TW');

    }

 

    function toggleAdvanced() {

      const panel = document.getElementById('advancedPanel');

      const arrow = document.getElementById('advArrow');

      const open = panel.classList.toggle('open');

      arrow.textContent = open ? '▲' : '▼';

    }

 

    function calc() {

      const input = parseFloat(document.getElementById('rent').value) || 0;

      const landlord = document.getElementById('landlordType').value;

      const taxMode = document.querySelector('input[name="taxMode"]:checked').value;

 

      const advOpen = document.getElementById('advancedPanel').classList.contains('open');

      let rate;

      if (advOpen) {

        rate = parseFloat(document.getElementById('taxRate').value);

      } else {

        rate = landlord === 'individual_foreign' ? 0.2 : 0.1;

      }

 

      const base = taxMode === 'included' ? input / (1 + rate) : input;

      const gross = taxMode === 'included' ? input : input * (1 + rate);

      const tax = base * rate;

 

      const nhiApply = landlord === 'individual_local' && base > 20000;

      const nhi = nhiApply ? base * 0.0211 : 0;

 

      const totalDeduct = tax + nhi;

      const net = base - nhi;

 

      const titles = {

        individual_local: '本國人',

        individual_foreign: '外國人',

        company: '公司'

      };

 

      document.getElementById('personTitle').textContent = `房東 1：${titles[landlord]}`;

      document.getElementById('rateLabel').textContent = (rate * 100).toFixed(0) + '%';

      document.getElementById('rentGross').textContent = fmt(gross);

      document.getElementById('taxAmt').textContent = fmt(tax);

      document.getElementById('nhiAmt').textContent = fmt(nhi);

      document.getElementById('nhiRow').style.opacity = nhiApply ? '1' : '0.4';

      document.getElementById('totalDeduct').textContent = fmt(totalDeduct);

      document.getElementById('netRent').textContent = fmt(net);

 

      const note = document.getElementById('deadlineNote');

      if (landlord === 'individual_local') {

        note.textContent = '本國人應於次月 10 日前繳納繳款書';

        note.style.display = 'flex';

      } else if (landlord === 'individual_foreign') {

        note.textContent = '非居住者應於給付時扣繳並於 10 日內繳納';

        note.style.display = 'flex';

      } else {

        note.style.display = 'none';

      }

    }

 

    calc();

  </script>
